5. The music signal has frequency components in the audio frequency range 20Hz to 20000Hz of the electromagnetic spectrum. Record or use the recorded music samples of different instrument.


clc;
clear;
close all;

% Load an audio file (replace with your own .wav file)
[instrument, Fs] = audioread('piano.wav');   % Example: piano
instrument = instrument(:,1);                % Use one channel if stereo

% Time vector for plotting
t = (0:length(instrument)-1)/Fs;

% Plot time-domain signal
figure;

subplot(3,1,1);
plot(t, instrument);
title('Time-Domain Signal');
xlabel('Time (s)');
ylabel('Amplitude');
grid on;

% Compute FFT
N = length(instrument);
f = (-N/2:N/2-1) * (Fs/N);
X = fftshift(fft(instrument));

% Amplitude Spectrum
amplitude = abs(X);

% Frequency vector
subplot(3,1,2);
plot(f, amplitude);
title('Amplitude Spectrum');
xlabel('Frequency (Hz)');
ylabel('|X(f)|');
xlim([0 Fs/2]);    % Focus on positive frequencies
grid on;

% Phase Spectrum
phase = angle(X);

subplot(3,1,3);
plot(f, phase);
title('Phase Spectrum');
xlabel('Frequency (Hz)');
ylabel('Phase (radians)');
xlim([0 Fs/2]);
grid on;
