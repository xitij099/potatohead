clc;
clear;
close all;

% Load audio file
[instrument, Fs] = audioread('piano.wav');

% If stereo, use only one channel
instrument = instrument(:,1);

% Time vector
t = (0:length(instrument)-1) / Fs;

% Plot Time-Domain Signal
figure;

subplot(3,1,1);
plot(t, instrument);
title('Time-Domain Signal');
xlabel('Time (s)');
ylabel('Amplitude');
grid on;

% FFT Computation
N = length(instrument);

% Frequency vector
f = (-N/2:N/2-1) * (Fs/N);

% FFT and shift zero frequency to center
X = fftshift(fft(instrument));

% Amplitude Spectrum
amplitude = abs(X);

subplot(3,1,2);
plot(f, amplitude);
title('Amplitude Spectrum');
xlabel('Frequency (Hz)');
ylabel('|X(f)|');
xlim([0 Fs/2]);   % Positive frequencies only
grid on;

% Phase Spectrum
phase = angle(X);

subplot(3,1,3);
plot(f, phase);
title('Phase Spectrum');
xlabel('Frequency (Hz)');
ylabel('Phase (radians)');
xlim([0 Fs/2]);
grid on;
